## 实现代码

### 功能

#### insert    (插入 (不断向上调))

```c++

void up(int k) //把第 k 个节点往上调
{
    while(k > 1)
    {
        int i = k / 2;

        if (h[k] < h[i])
        {
            swap(h[i], h[k]);
            k = i;
        }
        else
        {
            break;
        }
    }
} // log(n)

```

#### delete    (删除)

```c++

void down(int k) //把第 k 个节点往下调
{
    while(k + k < n)
    {
        int i = k + k;

        if (i + 1 <= n && h[i + 1] < h[i])
        {
            i++;
        }

        if (h[i] < h[k])
        {
            swap(h[i], h[k]);
            k = i;
        }
        else
        {
            break;
        }
    }
}

```

#### 其余功能

```c++

void add(int x) //添加
{
    ++n;
    h[n] = x;
    up(n);
}

void del(int pos) //删除下标为 pos 的元素
{
    swap(h[pos], h[n]);
    --n;
    up(pos);
    down(pos);
}

void del_v(int x) //删除值为 x 的元素
{
    int pos = find(x);

    if (pos)
    {
        del(pos);
    }
}
```
